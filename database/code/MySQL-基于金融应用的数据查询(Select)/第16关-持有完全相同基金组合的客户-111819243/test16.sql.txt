from property a,property b
where a.pro_c_id < b.pro_c_id and a.pro_type = 3 and b.pro_type = 3 and not exists
(
    select *
    from property c
    where c.pro_type = 3  and c.pro_c_id = b.pro_c_id and not exists
    (
        select *
        from property d
        where d.pro_type = 3 and d.pro_c_id = a.pro_c_id and d.pro_pif_id = c.pro_pif_id
    )
)
and not exists
(
