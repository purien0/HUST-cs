   -- 3) 查询购买了所有畅销理财产品的客户
--   请用一条SQL语句实现该查询：

select distinct a.pro_c_id
from property a
where not exists 
( select *
   from property b
   where b.pro_pif_id in 
(
   select distinct pro_pif_id
   from property
   where pro_type = 1
   group by pro_pif_id
   having count(distinct pro_c_id)>2
) and not exists 
(
   select * 
   from property c
   where c.pro_type = 1 and a.pro_c_id = c.pro_c_id and b.pro_pif_id = c.pro_pif_id
) 
)
order by pro_c_id asc;
