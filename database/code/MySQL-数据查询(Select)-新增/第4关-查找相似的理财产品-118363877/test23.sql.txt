select pro_pif_id,cc,prank
from
(
select pro_pif_id,cnt cc,dense_rank() over(order by cnt desc) as prank
from(
select pro_pif_id,count(*)cnt
from property
where pro_type = 1 and pro_pif_id in
(
select pro_pif_id#选出客户选的产品
from property
where pro_type = 1 and pro_pif_id != 14 and pro_c_id in
(
select pro_c_id#选出前三个客户
from
(
select pro_c_id,dense_rank() over(order by cnt desc) as ran_k
from
(
select pro_c_id,count(pro_id) cnt
from property
where pro_pif_id = 14 and pro_type = 1
group by pro_c_id
)as t

)as a
where ran_k<=3
)
)
group by pro_pif_id
)as c
)as d
where prank<=3
order by prank asc,pro_pif_id asc
