select distinct pro_c_id
from property
group by pro_c_id
having sum(case when pro_type = 1 then 1 else 0 end)>=3 and sum(case when pro_type = 1 then 1 else 0 end)>sum(case when pro_type = 3 then 1 else 0 end)
order by pro_c_id asc;
